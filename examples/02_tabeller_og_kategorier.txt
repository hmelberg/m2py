// Eksempel 2: Tabeller og kategorier
// Frekvenstabeller, kodelister og etiketter.

create-dataset tabeller
import fd/BEFOLKNING_KJOENN as kjonn
import fd/INNTEKT_WLONN 2022-01-01 as inntekt
import fd/BOSATT_KOMMUNE as kommune

define-labels kjonn_etikett 1 Mann 2 Kvinne
assign-labels kjonn kjonn_etikett

tabulate kjonn
tabulate kjonn, rowpct
tabulate kommune, top(10)

aggregate (mean) inntekt -> snitt_kommune, by(kommune)
tabulate kommune, summarize(inntekt) top(15)
